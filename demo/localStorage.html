本地存储

调用

                    
/**
 * 设置缓存
 * @key  {String} key - 缓存的key
 * @value  {Object} value - 缓存值，不设置则删除缓存内容
 * @duration  {Number} duration - 过期时间（秒），不设置或值小于1则为永久保存   
 */
KM.cache.set(key,value,duration);

/**
 * 获取缓存
 * @key  {String} key - 缓存的key      
 */
KM.cache.get(key);


/**
 * 移除缓存
 * @key  {String} key - 缓存的key     
 */
KM.cache.remove(key);


/**
 * 清空缓存      
 */
KM.cache.clear();
